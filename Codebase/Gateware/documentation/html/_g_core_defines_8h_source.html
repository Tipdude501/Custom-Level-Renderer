123  // Define the template class expander GProxy.
124  // This is a core pillar used in the C++11 modernized version of Gateware.
125  // A GProxy is a smart wrapper around a Gateware Interface & Implementation.
126  // The proxy allows safe transfer and allocation/deallocation of interfaces as well as safe access.
127  // Proxy rules: Copying creates safe handles but does not transfer ownership. Only moving r-values can transfer ownership.
128  template<typename _Interface, typename _Implementation, typename... _CreationParams>
129  class GProxy
130  {
131  protected:
132  typedef _Interface forceRaw; // used to support the rare need of construction from raw pointers (prefer to avoid)
133  typedef _Implementation forceStatic; // used to support static calls from proxy ops
134  typedef std::shared_ptr<const _Interface> forceConst; // used to support const correctness in proxy ops
135  typedef GProxy<_Interface, _Implementation, _CreationParams...> thisClass; // short hand for this type of class
136  std::weak_ptr<_Interface> access; // most common way of access via proxy, does not require ownership
137  std::shared_ptr<_Interface> storage; // Hard handle to data, not all instances use this, most are weak by default
138 
139  private:
140  // All Proxys are friends of one another (required for casting)
141  template<typename A, typename B, typename... C> friend class GProxy;
142  // This function is what actually makes a GProxy "proxy" an interface & associated implementation.
143  template<typename... _anyArgs>
144  GReturn allocate(_anyArgs&&... _argTransfer)
145  {
146  GReturn result = GReturn::FAILURE;
147  access.reset(); // clear any prior access, storage will be overwritten below
148  storage = std::make_shared<_Implementation>(); // so we can write the below more safely...
149  // *** A COMPILER ERROR HERE MEANS YOU ARE NOT PASSING ".Create(" A VALID SET OF ARGUMENTS ***
150  // *** READ OUR INCLUDED DOCUMENTATION TO FIND THE PROPER ARGUMENTS TO ANY GATEWARE FUNCTION ***
151  if (storage && G_PASS(result = dynamic_cast<_Implementation*>
152  (storage.get())->Create(std::forward<_anyArgs>(_argTransfer)...)))
153  access = storage; // used for primary access
154  else // if we failed to create, destruct the interface immediately
155  storage.reset();
156  // return what happened
157  return result;
158  }
159 
160  public:
161  // Creates an invalid but safe handle
162  GProxy() = default;
163  // Special constructor, allows creation from existing implementation.
164  // This should only be used in conjunction with std::shared_from_this from inside an implementation.
165  GProxy(const std::shared_ptr<_Interface>& _sharedFromThis)
166  {
167  access = storage = _sharedFromThis;
168  }
169  // copy constructor
170  GProxy(const thisClass& _cpy)
171  {
172  if (this != &_cpy)
173  access = _cpy.access;
174  }
175  // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
176  // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
177  template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
178  GProxy(const T& _cpy)
179  {
180  access = std::dynamic_pointer_cast<_Interface>(_cpy.access.lock());
181  }
182  // move constructor does a direct transfer since it is not considered a copy
183  GProxy(thisClass&& _mov) noexcept //: storage(_mov.storage), access(_mov.access)
184  {
185  storage = std::move(_mov.storage);
186  access = std::move(_mov.access);
187  }
188  // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
189  // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
190  template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
191  GProxy(T&& _mov) noexcept //: storage(std::dynamic_pointer_cast<_Interface>(std::move(_mov.storage))), access(storage)
192  {
193  storage = std::dynamic_pointer_cast<_Interface>(std::move(_mov.storage));
194  access = (storage) ? storage : std::dynamic_pointer_cast<_Interface>(_mov.access.lock());
195  }
196  // Assignment operator
197  virtual thisClass& operator=(const thisClass& _cpy) final
198  {
199  if (this != &_cpy)
200  {
201  access = _cpy.access;
202  storage.reset(); // copy assignment may not transfer or retain any prior ownership
203  }
204  return *this;
205  }
206  // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
207  // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
208  template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
209  thisClass& operator=(const T& _cpy)
210  {
211  access = std::dynamic_pointer_cast<_Interface>(_cpy.access.lock());
212  storage.reset(); // copy assignment may not transfer or retain any prior ownership
213  return *this;
214  }
215  // move assignment
216  virtual thisClass& operator=(thisClass&& _mov) noexcept final
217  {
218  storage = std::move(_mov.storage);
219  access = std::move(_mov.access);
220  return *this;
221  }
222  // template version, attempts to convert incoming data to this class type using dynamic_pointer_cast if possible
223  // We use SFINAE to ensure this is only called for non-derived types, those should use the standard overloads
224  template<typename T, typename = std::enable_if<!std::is_convertible<T*, thisClass*>::value>>
225  thisClass& operator=(T&& _mov) noexcept
226  {
227  storage = std::dynamic_pointer_cast<_Interface>(std::move(_mov.storage));
228  access = (storage) ? storage : std::dynamic_pointer_cast<_Interface>(_mov.access.lock());
229  return *this;
230  }
231  // basic destructor
232  ~GProxy() = default;
233  // Allows testing the class for validity
234  virtual operator bool() const final { return !access.expired(); };
235  // nullptr assignment is how to clear a proxy before waiting for it to fall out of scope
236  virtual thisClass& operator =(std::nullptr_t) final { access.reset(); storage.reset(); return *this; };
237  // The GProxy varadic template should match the main internal implementation "Create" function.
238  // Though the templated variadic version of this function can support any set of arguments,
239  // This version requires derived classes to define & support at least one specific Create variant.
240  // This is required for API consistency. Alternate Create routines are supported by the next function.
241  virtual GReturn Create(_CreationParams... _parameter) final
242  {
243  return allocate<_CreationParams...>(std::forward<_CreationParams>(_parameter)...);
244  }

246  // This version is much more functional but cannot be virtual which is why the prior version exists.
247  template< typename... _AltArguments,
248  typename = std::enable_if<!std::is_same< // SFINAE used to force virtual override in impl.
249  std::tuple<_AltArguments...>, std::tuple<_CreationParams...> >::value>>
250  GReturn Create(_AltArguments... _parameter)
251  {
252  return allocate<_AltArguments...>(std::forward<_AltArguments>(_parameter)...);
253  }
254  // Return a copy of the Proxy that allows for non-weak transfer of ownership
255  // If direct ownership cannot be shared it will return an invalid handle
256  virtual thisClass Share() const final
257  {
258  thisClass canShare;
259  if (storage)
260  {
261  canShare.storage = storage;
262  canShare.access = access;
263  }
264  return canShare; // r-value will invoke move to allow full transfer
265  }
266  // Surrenders strong ownership of your internal handle to another Proxy, weak ownership is maintained
267  // If direct ownership cannot be surrendered it will return an invalid handle
268  virtual thisClass Relinquish() final
269  {
270  thisClass canYield;
271  if (storage)
272  {
273  canYield.storage = storage;
274  canYield.access = access;
275  storage.reset();
276  }
277  return canYield;
278  }
279  // Returns a direct copy of the Proxy with the current ownership status maintined
280  // If direct ownership cannot be shared it will provide it's weak ownership
281  virtual thisClass Mirror() const final
282  {
283  thisClass canReflect;
284  canReflect.storage = storage;
285  canReflect.access = access;
286  return canReflect;
287  }

404  // Defines the simple type-erasure struct GEvent
405  // This is a another core pillar used in the C++11 modernized version of Gateware.
406  // A GEvent is a generic but type-safe way to communicate small-footprint event messages to listeners.
407  // All sent messages must fit within G_MESSAGE_BLOCK_SIZE total including type_info hashes.
408  // A GEvent only supports safe transfer of raw data blocks(C structs, POD) & enum class types.
409  // (Copy)Constructors/Destructors/Assignment Operators etc... of stored types are NOT respected in any way!
410  // Future versions (c++17+) of Gateware could replace some of this with the FAR more robust std::any if needed.
411 
412  //! Typesafe and Typeless communication of simplistic event data. Used by GEventGenerator & GEventReceiver. (Observer pattern)
413  /*!
414  * This struct is how GEventGenerators communicate custom Events to any GEventReceiver who registers themselves to the object.
415  * This struct alongside std::function/std::bind is used to circumvent the traditional derivation based observer model.
416  * You can think of GEvent as a less-robust and more specialized version of std::any. (we currently only target c++11)
417  */
418  struct GEvent
419  {
420  private:
421  std::size_t enumType = 0; // identifies stored enumerator representing the event ID
422  std::size_t dataType = 0; // identifies stored structure representing the associated data bundled with an event
423  // raw data containing specific enum value & data packet.
424  // The leading byte of each type-erased data block records the size of the embedded data.
425  // This information is used for additional data integrity verification during reads and offset jumps when required.
426  unsigned char rawData[G_RAW_DATA_PACKET_SIZE] = {0,};
427  // due to odd include order issues across platforms regarding std::memcpy & std::copy
428  // we provide a substitution here that copies a fixed amount of bytes
429  static inline void memory_copy(void *_dst, const void* _src, unsigned int _byteCount)
430  {
431  std::size_t* write = static_cast<std::size_t*>(_dst);
432  std::size_t const *read = static_cast<std::size_t const*>(_src);
433  for (; _byteCount >= sizeof(std::size_t); _byteCount -= sizeof(std::size_t))
434  *write++ = *read++;
435  unsigned char* w = reinterpret_cast<unsigned char*>(write);
436  unsigned char const* r = reinterpret_cast<unsigned char const*>(read);
437  for (; _byteCount; --_byteCount)
438  *w++ = *r++;
439  }
440 
441  public:
442  //! Allows writing of any enum class & raw data to a GEvent while retaining internal type hashes.(typeerasure)
443  /*!
444  * DEVNOTE: Written event data must be <= to (G_MESSAGE_BLOCK_SIZE - 16) - sizeof(chosen enum class type).
445  * Only base type enums & raw data streams (C style structs) may be written as data (POD Plain Old Data)
446  * Each stored type may not exceed 255 bytes for size verification/offset reasons.
447  *
448  * \param _eventEnumValue Enum used to represent which event has occured.(any simple POD enum valid)
449  * \param _eventData Extra data that acompanies this event.(any small POD works here)
450  *
451  * \retval SUCCESS Valid types were provided and the event data was successfully embedded.
452  * \retval UNEXPECTED_RESULT Encoded type matched the provided type hash but does not match its size in bytes.
453  */
454  template<class enumT, typename dataT>
455  GReturn Write(const enumT& _eventEnumValue, const dataT& _eventData)
456  {
457  // ensure developers stay within data size limits of storable data types
458  static_assert( sizeof(enumT) < 256, "Enum types stored by GEvent must be < 256 bytes!");
459  static_assert( sizeof(dataT) < 256, "Data types stored by GEvent must be < 256 bytes!");
460  // ensure developers do not exceed internal storage limits
461  static_assert( (sizeof(enumT) + sizeof(dataT)) <= G_RAW_DATA_PACKET_SIZE,
462  "Enum + Data combined size attempted to be stored by GEvent is too large!");
463  // ensure developers do not send types other than an enum to identify their events
464  static_assert( std::is_enum<enumT>::value,
465  "Gateware Events{} are required to use an \"enum class\" list to identify themselves!");
466  // ensure developers use only C Style stuctures & basic enumerators for information transfer
467  static_assert( std::is_trivially_copyable<dataT>::value,
468  "Enum & Data Types must be fully tranferable with std:memcpy only! Plain Old Data Only!");
469  // Once we know the storable types are safe we record their type and transfer the data
470  enumType = typeid(enumT).hash_code();
471  dataType = typeid(dataT).hash_code();
472  // Unlikely, but the standard doesn't say it can't happen, would interfere with error checking
473  if (enumType == 0 || dataType == 0)
474  return GReturn::UNEXPECTED_RESULT;
475  // Store bit data for typeless transfer
476  rawData[0] = sizeof(enumT); // encode size of enum
477  memory_copy(rawData + 1, &_eventEnumValue, sizeof(enumT)); // store enum
478  rawData[sizeof(enumT) + 1] = sizeof(dataT); // encode size of data
479  memory_copy(rawData + sizeof(enumT) + 2, &_eventData, sizeof(dataT)); // store data
480  // transfer to internal storage should now be successful
481  return GReturn::SUCCESS;
482  };
483  //! Safely extract G::Events and G::EVENT_DATA from a GEventGenerator if you know what type of information you need.
484  /*!
485  * Once your GEventReceiver has responded to an event and is receiving a GEvent, you can use this operation
486  * to safely extract information about a specific event. Not only can you establish which event was generated,
487  * you can also receive detailed information about the event if that event contains additional data.
488  *
489  * \param [out] _outEventID Variable used to capture which event has occured.(See interface docs)
490  * \param [out] _outEventData Variable used to capture incoming data from an event.(See interface docs)
491  *
492  * \retval SUCCESS The proper types were provided and the event data was successfully extracted.
493  * \retval FAILURE This GEvent was never intialized with any event data. (Dataless Event)
494  * \retval INVALID_ARGUMENT Provided template data types do NOT match internally stored hashes for this event.
495  * \retval MEMORY_CORRUPTION Encoded type matched the provided type hash but does not match its size in bytes.
496  */
497  template<class enumT, typename dataT>
498  GReturn Read(enumT& _outEventID, dataT& _outEventData) const
499  {
500  // If you get this complier error you are providing Read() with an incompatible type.
501  // Always use Gxxx::Events and Gxxx::EVENT_DATA to examine an event.
502  static_assert((sizeof(enumT) + sizeof(dataT)) <= G_RAW_DATA_PACKET_SIZE &&
503  std::is_enum<enumT>::value && std::is_trivially_copyable<enumT>::value &&
504  std::is_trivially_copyable<dataT>::value,
505  "Types provided to GEvent::Read() operation are not a compatible types!"
506  "Use appropriate GEventGenerator derived G::Events enum and G::EVENT_DATA structure.");
507  // Type hashes will ensure safe access of underlying data in 99.9% of situations
508  // Not 100% perfect but much safer than a void* like it used to be.
509  if (enumType == 0 || dataType == 0) return GReturn::FAILURE; // No event data present
510  // Requested type does not match internal enum format (use proper G::Events)
511  if (enumType != typeid(enumT).hash_code()) return GReturn::INVALID_ARGUMENT;
512  // Requested type does not match internal data structure format (use proper G::EVENT_DATA)
513  if (dataType != typeid(dataT).hash_code()) return GReturn::INVALID_ARGUMENT;
514  // Once types have been safely checked we verify the byte sizes of types and copy data
515  if (sizeof(enumT) == rawData[0])
516  memory_copy(&_outEventID, rawData + 1, sizeof(enumT));
517  else
518  return GReturn::MEMORY_CORRUPTION; // data size verification failed
519  // check that data type is also size verified before transfer
520  if (sizeof(dataT) == rawData[sizeof(enumT) + 1])
521  memory_copy(&_outEventData, rawData + sizeof(enumT) + 2, sizeof(dataT));
522  else
523  return GReturn::MEMORY_CORRUPTION; // data size verification failed
524  // if we made it this far everything checked out and transfered successfully.
525  return GReturn::SUCCESS;
526  }
527  //! Safely extract G::Events and G::EVENT_DATA from a GEventGenerator if you know what type of information you need.
528  /*!
529  * Once your GEventReceiver has responded to an event and is receiving a GEvent, you can use this operation
530  * to safely extract information about a specific event. Not only can you establish which event was generated,
531  * you can also receive detailed information about the event if that event contains additional data.
532  *
533  * \param [out] _outEventOrData Variable used to capture incoming data or event ID from an event.(See interface docs)
534  *
535  * \retval SUCCESS A proper type was provided and the information was successfully extracted.
536  * \retval FAILURE This GEvent was never intialized with any information. (Uninitialized Event)
537  * \retval INVALID_ARGUMENT Provided template data type does NOT match any internal stored hash for this event.
538  * \retval MEMORY_CORRUPTION Encoded type matched the provided type hash but does not match its size in bytes.
539  */
540  template<class enumOrDataT>
541  GReturn Read(enumOrDataT& _outEventOrData) const
542  {
543  // If you get this complier error you are providing Read() with an incompatible type.
544  // Always use Gxxx::Events and Gxxx::EVENT_DATA to examine an event.
545  static_assert(sizeof(enumOrDataT) <= G_RAW_DATA_PACKET_SIZE &&
546  std::is_trivially_copyable<enumOrDataT>::value,
547  "Type provided to GEvent::Read() operation is not a compatible type!"
548  "Use appropriate GEventGenerator derived G::Events enum or G::EVENT_DATA structure.");
549  // Type hashes will ensure safe access of underlying data in 99.9% of situations
550  // Not 100% perfect but much safer than a void* like it used to be.
551  if (enumType == 0 && dataType == 0)
552  return GReturn::FAILURE; // No event data present
553  // enum type identified
554  else if (enumType == typeid(enumOrDataT).hash_code())
555  if (sizeof(enumOrDataT) == rawData[0])
556  memory_copy(&_outEventOrData, rawData + 1, sizeof(enumOrDataT)); // transfer enum bits
557  else // data size verification failed
558  return GReturn::MEMORY_CORRUPTION;
559  // event data identified
560  else if (dataType == typeid(enumOrDataT).hash_code())
561  if (rawData[0] < (G_EVENT_BLOCK_SIZE - (sizeof(std::size_t) << 1)) && // enoded enum type integrity check
562  sizeof(enumOrDataT) == rawData[rawData[0] + 1]) // jump to type size offset
563  memory_copy(&_outEventOrData, rawData + rawData[0] + 2, sizeof(enumOrDataT)); // transfer enum bits
564  else // data size verification failed
565  return GReturn::MEMORY_CORRUPTION;
566  else // Requested type does not match internal enum/data structure format (use proper G::Events/G::EVENT_DATA)
567  return GReturn::INVALID_ARGUMENT;
568  // with transfer complete notify success
569  return GReturn::SUCCESS;
570  }
571  };
