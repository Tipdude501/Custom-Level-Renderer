166  //! Tile drawing options supported by GBlitter. Bitflags indicating which transformations should be applied to a tile during drawing.
167  //! Notes:
168  //! Eight basic orientations are available using combinations of the ROTATE, MIRROR_VERTICAL, and MIRROR_HORIZONTAL flags (all rotation angles are clockwise; "inverse" indicates mirroring about the tile's local y axis):
169  //! 0* = 0
170  //! 0* inverse = MIRROR_HORIZONTAL
171  //! 90* = ROTATE
172  //! 90* inverse = ROTATE | MIRROR_VERTICAL
173  //! 180* = MIRROR_VERTICAL | MIRROR_HORIZONTAL
174  //! 180* inverse = MIRROR_VERTICAL
175  //! 270* = ROTATE | MIRROR_VERTICAL | MIRROR_HORIZONTAL
176  //! 270* inverse = ROTATE | MIRROR_HORIZONTAL
177  enum DrawOptions
178  {
179  ROTATE = 0x0001, //!< Indicates that a tile should be rotated 90 degrees clockwise. (Only used by non-transformed instructions.)
180  MIRROR_VERTICAL = 0x0002, //!< Indicates that a tile should be flipped in the y axis. (Only used by non-transformed instructions.)
181  MIRROR_HORIZONTAL = 0x0004, //!< Indicates that a tile should be flipped in the x axis. (Only used by non-transformed instructions.)
182  IGNORE_SOURCE_COLORS = 0x0008, //!< Indicates that per-pixel color values from source should NOT be used. (I.E., only layer and/or stencil data from tile should be drawn to the canvas.)
183  USE_SOURCE_LAYERS = 0x0010, //!< Indicates that per-pixel layer values from source should be used.
184  USE_SOURCE_STENCILS = 0x0020, //!< Indicates that per-pixel stencil values from source should be used.
185  USE_MASKING = 0x0040, //!< Indicates that key-value bitmasking should be applied to a tile. Affects color, layer, and stencil data. (Opaque/transparent. Slightly less efficient than no transparency.)
186  USE_TRANSFORMATIONS = 0x0080, //!< Indicates that the transformation matrix and pivot should be applied to a tile. Note: Transformations use screen axes (I.E., -y is up, +y is down), NOT mathematical axes (ie, -y is down, +y is up).
187  USE_TRANSPARENCY = 0x0100, //!< Indicates that alpha-channel color blending should be applied to a tile. Affects color data. (Semi-transparency. Significantly less efficient than no transparency or bitmasking. Overuse will reduce performance.)
188  };

253  //! Loads and stores a read-only raster source to define tiles from.
254  /*!
255  * Use this function to load source data from files directly.
256  * All types of data are optional, but at least one must be used for a source to be valid. All images used must have the same dimensions.
257  * Greyscale (especially with no alpha channel) is recommended for layer/stencil data since the filesize will be significantly smaller.
258  * If a file does not contain an alpha channel, the alpha values will be set to maximum.
259  * Supported file types:
260  * TGA (Targa)
261  * Supported color formats:
262  * - 8 bits-per-pixel greyscale (Luminance) [TGA documentation refers to this as black-and-white]
263  * - 16 bits-per-pixel greyscale with transparency (Alpha-Luminance) [TGA documentation refers to this as black-and-white]
264  * - 24 bits-per-pixel RGB color (Red-Green-Blue) [TGA documentation refers to this as true-color]
265  * - 32 bits-per-pixel RGB color with transparency (Alpha-Red-Green-Blue) [TGA documentation refers to this as true-color]
266  * - 8 bits-per-pixel indexed color (Look-Up Table) [TGA documentation refers to this as color-mapped]
267  * -- Only 24-bit palette entries and 8-bit palette indices are supported.
268  * -- Be sure all colors in the palette are opaque, otherwise the palette will be read incorrectly.
269  * Maximum dimensions supported: 65,535 x 65,535
270  * (TGA files do not support anything larger than this, so attempting to export a TGA larger than this will corrupt the data. The library is not able to detect if this has happened except at exact multiples of 65,536 in either dimension.)
271  * Maximum sources supported: 65,535
272  *
273  * \param [in] _colorsFilepath (OPTIONAL) Path to the file to load color values from. (Pass nullptr if not needed.)
274  * \param [in] _layersFilepath (OPTIONAL) Path to the file to load layer values from. Only the red channel will be used if file is in a color format. (Pass nullptr if not needed.)
275  * \param [in] _stencilsFilepath (OPTIONAL) Path to the file to load stencil values from. Only the red channel will be used if file is in a color format. (Pass nullptr if not needed.)
276  * \param [out] _outIndex Index of the stored source.
277  *
278  * \retval GReturn::INVALID_ARGUMENT All filepaths were nullptr, one or more filepaths do not contain a file extension, or two or more images' dimensions did not match.
279  * \retval GReturn::FAILURE GFile proxy could not be created to read files or image data could not be read from a file.
280  * \retval GReturn::FILE_NOT_FOUND One or more filepaths do not point to a valid file.
281  * \retval GReturn::FORMAT_UNSUPPORTED One or more filepaths contains an extension that is not supported. (See supported file types above.)
282  * \retval GReturn::FEATURE_UNSUPPORTED One or more files has image format features that are not supported. (Make sure file settings are correct when exporting; See supported file types above.)
283  * \retval GReturn::IGNORED Maximum number of sources would be exceeded. (Consider combining sources together. If you need more than the maximum, you need to optimize.)
284  * \retval GReturn::SUCCESS Source was loaded and stored.
285  */
286  virtual GReturn LoadSource(const char* _colorsFilepath, const char* _layersFilepath, const char* _stencilsFilepath, unsigned short &_outIndex) = 0;
